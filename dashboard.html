0).toFixed(2); todayEl.innerText = "â‚¹" + (d.todayEarning ?? 0).toFixed(2); totalEl.innerText = "â‚¹" + (d.totalEarning ?? 0).toFixed(2); tasksEl.innerText = d.completedTasks ?? 0; loader.style.display = "none"; // ðŸ”¥ NOW IT WILL WORK } catch (e) { console.error(e); showToast("Dashboard load failed"); } }); /* LOGOUT */ logoutBtn.onclick = async () => { await signOut(auth); window.location.href = "index.html"; }; /* TASK CLICK */ document.querySelectorAll(".task-btn").forEach(btn => { btn.addEventListener("click", async () => { const reward = parseFloat(btn.dataset.reward); const url = btn.dataset.url; const user = auth.currentUser; if (!user) return; window.open(url, "_blank"); btn.disabled = true; let time = 30; const original = btn.innerText; const timer = setInterval(() => { btn.innerText = `Wait ${time}s`; time--; if (time < 0) clearInterval(timer); }, 1000); setTimeout(async () => { const ref = doc(db, "users", user.uid); await updateDoc(ref, { balance: increment(reward), todayEarning: increment(reward), totalEarning: increment(reward), completedTasks: increment(1) }); showToast(`â‚¹${reward} added`); btn.innerText = original; btn.disabled = false; }, 30000); }); });
